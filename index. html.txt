  <script>
    // Builder Pattern
    class Pizza {
      constructor(size, toppings, addon, delivery) {
        this.size = size;
        this.toppings = toppings;
        this.addon = addon;
        this.delivery = delivery;
      }
    }

    class PizzaBuilder {
      constructor() {
        this.size = "Medium";
        this.toppings = [];
        this.addon = "None";
        this.delivery = "Cash on Delivery";
      }

      setSize(size) { this.size = size; return this; }
      addToppings(toppings) { this.toppings = toppings; return this; }
      setAddon(addon) { this.addon = addon; return this; }
      setDelivery(delivery) { this.delivery = delivery; return this; }
      build() {
        return new Pizza(this.size, this.toppings, this.addon, this.delivery);
      }
    }

    function buildPizza() {
      let size = document.getElementById("size").value;
      let addon = document.getElementById("addon").value;
      let delivery = document.getElementById("delivery").value;
      let toppings = [];
      document.querySelectorAll("input[type=checkbox]:checked").forEach(c => toppings.push(c.value));

      // Use Builder
      let pizza = new PizzaBuilder()
        .setSize(size)
        .addToppings(toppings)
        .setAddon(addon)
        .setDelivery(delivery)
        .build();

      document.getElementById("result").style.display = "block";
      document.getElementById("result").innerHTML = `
        <strong>‚úÖ Order Summary:</strong><br>
        üçï Size: <b>${pizza.size}</b><br>
        üßÄ Toppings: ${pizza.toppings.length ? pizza.toppings.join(", ") : "None"}<br>
        ‚ûï Add-on: ${pizza.addon}<br>
        üí≥ Payment Method: ${pizza.delivery}
      `;
    }
  </script>
